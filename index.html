<script>
    async function processImage() {
        const fileInput = document.getElementById('passportImage');
        const statusDiv = document.getElementById('status');
        const extractBtn = document.getElementById('extractBtn');
        const downloadBtn = document.getElementById('downloadBtn');

        if (fileInput.files.length === 0) {
            alert("الرجاء اختيار صورة جواز السفر أولاً.");
            return;
        }

        const file = fileInput.files[0];
        
        // تغيير حالة الواجهة
        extractBtn.disabled = true;
        downloadBtn.style.display = "none";
        statusDiv.innerText = "جاري قراءة الصورة وتفكيك الشفرة (MRZ)... يرجى الانتظار.";

        try {
            // استخدام Tesseract لقراءة النص من الصورة
            const result = await Tesseract.recognize(file, 'eng', {
                logger: m => {
                    if(m.status === 'recognizing text'){
                        statusDiv.innerText = `جاري التحليل: ${Math.round(m.progress * 100)}%`;
                    }
                }
            });

            const text = result.data.text;
            
            // البحث عن سطور الـ MRZ (التي تحتوي على علامات المتباينة <<<<)
            const lines = text.split('\n');
            const mrzLines = lines.filter(line => line.includes('<<'));

            if (mrzLines.length >= 2) {
                statusDiv.innerText = "✅ تم استخراج البيانات بنجاح!";
                statusDiv.style.color = "#27ae60";
                
                // تجهيز النص ليتم تحميله
                // ملاحظة: هنا قمنا بسحب السطور المشفرة كاملة، يمكنك برمجياً تفكيكها لاحقاً
                let txtContent = "=== بيانات جواز السفر المستخرجة ===\n\n";
                txtContent += "الشفرة المقروءة (MRZ):\n";
                mrzLines.forEach(line => {
                    txtContent += line.replace(/\s/g, '') + "\n"; // تنظيف المسافات العشوائية
                });

                // إنشاء ملف .txt داخل المتصفح
                const blob = new Blob([txtContent], { type: "text/plain;charset=utf-8" });
                const url = URL.createObjectURL(blob);
                
                // تفعيل زر التحميل
                downloadBtn.href = url;
                downloadBtn.download = "Passport_Data.txt";
                downloadBtn.style.display = "inline-block";

            } else {
                statusDiv.innerText = "❌ لم يتم العثور على شفرة الجواز (MRZ). تأكد من وضوح الصورة أسفل الجواز.";
                statusDiv.style.color = "#e74c3c";
            }

        } catch (error) {
            statusDiv.innerText = "حدث خطأ أثناء القراءة.";
            console.error(error);
        } finally {
            extractBtn.disabled = false;
        }
    }
</script>
